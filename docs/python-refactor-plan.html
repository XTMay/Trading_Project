1 执行摘要

提示一句话总结

将盈再表从 Windows-only VBA 单体系统 迁移为 跨平台 Python 三层架构，以 yfinance 为核心数据源，用 pandas 替代 Excel 单元格计算，用 openpyxl/xlwings 保持 Excel 兼容，最终实现可维护、可测试、跨平台的现代化投资分析系统。

1.1 迁移动机

现状痛点 Python 方案
完全依赖 Windows COM 组件（MSXML2、WinHttp、IE） requests + yfinance — 跨平台
10+ 个脆弱的 HTML 抓取数据源 yfinance API — 单一稳定接口
On Error Resume Next 静默吞错 Python 结构化异常 + logging
90+ 函数分散在 9 个 Module 中，重复代码严重 OOP 类封装 + 职责分离
所有 URL / 列号硬编码在代码中 配置文件（YAML / .env）外部化
无法在 macOS 运行 Python 天然跨平台
无单元测试 pytest 全覆盖

1.2 迁移原则

  1. 渐进式迁移 — VBA 和 Python 共存过渡，不一次性全部重写
  2. 数据优先 — 先迁移数据抓取层（价值最高、风险最大的部分）
  3. API 优于抓取 — 能用 API 的绝不再 scrape HTML
  4. Excel 兼容 — 保留 .xlsm 作为 UI 前端，Python 作为后端引擎
  5. 可测试 — 每个模块都有对应的单元测试

7 反爬策略

重要为什么需要反爬策略？

VBA 原系统使用 On Error Resume Next 忽略所有网络错误。 一旦被目标网站限速或封锁，数据静默缺失，用户完全不知情。 Python 系统必须主动应对反爬机制，确保数据可靠性。

8.1 数据层：YFinanceProvider

# src/yinzaibiao/data_providers/yfinance_provider.py

from dataclasses import dataclass
import yfinance as yf
import pandas as pd

# 交易所后缀映射（替代 VBA yhondd 函数）
EXCHANGE_SUFFIX = {
    "TW": ".TW",   "TWO": ".TWO",  # 台股上市/上柜
    "HK": ".HK",                    # 港股
    "CN_SH": ".SS", "CN_SZ": ".SZ", # 中股
    "JP": ".T",    "UK": ".L",      # 日/英
    "AU": ".AX",   "KR": ".KS",     # 澳/韩
    "FR": ".PA",   "CA": ".TO",     # 法/加
    "US": "",                        # 美股无后缀
}

class YFinanceProvider:
    """统一数据提供者 — 替代 Module2~5 的数据抓取逻辑。"""

    def __init__(self, cache: CacheManager, http: HttpClient):
        self.cache = cache
        self.http = http

    def get_ticker(self, code: str, market: str) -> yf.Ticker:
        suffix = EXCHANGE_SUFFIX.get(market, "")
        return yf.Ticker(f"{code}{suffix}")

    def get_history(self, code: str, market: str,
                    period: str = "10y") -> pd.DataFrame:
        """替代: Yahoo_Price_Dividend_Split + highlow + splitt。
        yfinance 自动返回拆股调整价格，无需手动调整。"""
        ticker = self.get_ticker(code, market)
        return ticker.history(period=period)

    def get_financials(self, code: str, market: str) -> dict:
        """替代: ConnectMarketWatch 的财报抓取。
        返回 {income_stmt, balance_sheet, cashflow} 的 DataFrame 字典。"""
        ticker = self.get_ticker(code, market)
        return {
            "income_stmt": ticker.income_stmt,
            "quarterly_income_stmt": ticker.quarterly_income_stmt,
            "balance_sheet": ticker.balance_sheet,
            "quarterly_balance_sheet": ticker.quarterly_balance_sheet,
            "cashflow": ticker.cashflow,
        }

    def get_dividends(self, code: str, market: str) -> pd.Series:
        """替代: Yahoo CSV 股息下载。"""
        return self.get_ticker(code, market).dividends

    def get_info(self, code: str, market: str) -> dict:
        """替代: ConnectMarketWatch 公司概况 + ConnectWinHttp 市值。"""
        return self.get_ticker(code, market).info

    def get_fx_rate(self, from_ccy: str, to_ccy: str = "TWD") -> float:
        """替代: ConnectWinHttp(wise_url) 汇率抓取。"""
        pair = f"{from_ccy}{to_ccy}=X"
        ticker = yf.Ticker(pair)
        hist = ticker.history(period="1d")
        return hist["Close"].iloc[-1]

8.2 业务层：PerformanceTracker

# src/yinzaibiao/analyzers/performance.py

import numpy as np
from scipy.optimize import brentq

class PerformanceTracker:
    """替代 Module8 — 绩效追踪 + 巴菲特距离。"""

    BUFFETT_ANNUAL_RETURN = 0.20   # 巴菲特 50 年 20%
    BUFFETT_YEARS = 50
    BUFFETT_MULTIPLIER = 1.20 ** 50  # ≈ 9,100
    NORMALIZE_BASE = 3_701          # 8年12%基准

    def calculate_xirr(self, cashflows: list[tuple]) -> float:
        """替代 VBA 的 Excel XIRR 公式。
        cashflows: [(date, amount), ...] — 负数=买入，正数=卖出/现值
        """
        dates, amounts = zip(*cashflows)
        # 转换为年份差
        d0 = min(dates)
        years = [(d - d0).days / 365.25 for d in dates]

        def npv(rate):
            return sum(a / (1 + rate) ** y
                      for a, y in zip(amounts, years))

        return brentq(npv, -0.99, 10.0)

    def buffett_distance(self, annual_return: float,
                         invest_years: float) -> float:
        """替代 VBA cpbf() — 巴菲特距离公式。
        Distance = 100 × √A × B / 3701
        A = 1.2^(50 - invest_years)
        B = log(9100) / log(1 + annual_return)
        """
        A = 1.2 ** (self.BUFFETT_YEARS - invest_years)
        if annual_return <= 0:
            return float('inf')
        B = np.log(self.BUFFETT_MULTIPLIER) / np.log(1 + annual_return)
        return 100 * np.sqrt(A) * B / self.NORMALIZE_BASE

    def evaluate(self, distance: float) -> str:
        """巴菲特距离评级。"""
        if distance <= 21:
            return "堪比巴菲特！"
        elif distance <= 100:
            return "太棒了！"
        elif distance <= 168:
            return "加油！"
        else:
            return "常读讲义"

8.3 业务层：ValuationEngine

# src/yinzaibiao/analyzers/valuation.py

class ValuationEngine:
    """替代 Module7.w2k 中内联的估值逻辑。"""

    HOLD_YEARS = 8  # 投资年限假设

    def expected_annual_return(self, base_return: float,
                                current_price: float,
                                reference_price: float) -> float:
        """预期年化报酬率 = 基础报酬率 - [(当前价/基准价)^(1/8) - 1]"""
        price_growth = (current_price / reference_price) ** (
            1 / self.HOLD_YEARS) - 1
        return base_return - price_growth

    def price_zone(self, expected_return: float) -> str:
        """便宜价/合理价/昂贵价 三区间判断。"""
        if expected_return > 0.15:
            return "便宜（买入区）"
        elif expected_return > 0.08:
            return "合理（持有区）"
        else:
            return "昂贵（卖出区）"
